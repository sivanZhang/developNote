//HTML
<input type="text" id="demo"/>
<span>0/50</span>

//CSS
.ta-wrap span {
    position: absolute;
    right: 20px;
    bottom: 8px;
    color: #AAA;
}

//JS
$("#demo").on('keyup input', function (event) {
    var val = $(this).val();
    var len = val.length;
    var count = $(this).siblings('span');

    if (len == 0) { count.text("0/50"); return; }
    if (len > 50) {
        len = 50;
        $(this).val(val.substring(0, 50));
    }
    count.text(len + "/50");
});
